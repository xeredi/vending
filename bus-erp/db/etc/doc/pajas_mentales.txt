SELECT
    round(extract('epoch' from lgps_fecha) / 300) * 300 as timestamp
    , lgps_vhcl_pk
    , count(lgps_vhcl_pk)
    , max(lgps_pk)
    , max(lgps_spd)
    , min(lgps_spd)
    , avg(lgps_spd)
    , sum(lgps_dst)
    , sum(lgps_elt)
FROM tbl_lectura_gps_lgps
WHERE
	lgps_fecha >= '2018-01-17 00:00:00'
	AND lgps_fecha < '2018-01-17 01:30:00'
	AND lgps_vhcl_pk = 1700
GROUP BY lgps_vhcl_pk, round(extract('epoch' from lgps_fecha) / 300)
;
